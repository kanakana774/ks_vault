```SQL
DO $$
DECLARE
    target_customer_id INTEGER:= 1;
	new_email TEXT:= 'new.email@example.com';
	affected_rows INTEGER;
    sql_query TEXT;

BEGIN

 sql_query:= 'UPDATE customer SET email = '' ' || new_email || ' '' WHERE customer_id = $1';
	EXECUTE sql_query USING  target_customer_id;
	GET DIAGNOSTICS affected_rows = ROW_COUNT;

	RAISE NOTICE '更新された顧客のID% 新しいメールアドレス% 影響を受けた行数%',target_customer_id,new_email,affected_rows;

END $$;
```

```SQL
sql_query:= 'UPDATE customer SET email = '' ' || new_email || ' '' WHERE customer_id = $1';
```

```SQL
'UPDATE customer SET email = '' '
```

```SQL
UPDATE customer SET email = '
```
